<!-- Catalog entry schema — verbatim from CATALOG-02. -->

interface CatalogEntry {
  id: string;            // stable across builds for the same element (hash of kind + i18nKey + imageRef + domPath)
  kind: 'i18n-text' | 'i18n-html' | 'image' | 'gallery-image' | 'heading' | 'hardcoded-text';

  // locator-signal fields (subset populated per kind):
  i18nKey?: string | null;
  i18nAttr?: 'data-i18n' | 'data-i18n-html' | 'data-i18n-placeholder' | null;
  currentText?: string | null;
  currentTextFr?: string | null;      // populated only for FR-locale route variants (out of scope for v1.3 — leave undefined)
  nearestHeading?: string | null;
  domPath?: string | null;
  imageRef?: string | null;
  altText?: string | null;            // for kind: 'image' or 'gallery-image'
  galleryAttrRaw?: string | null;     // for kind: 'gallery-image' — the data-gallery / data-room ancestor value
  galleryIndex?: number | null;       // for kind: 'gallery-image' — 0-based position inside the gallery

  // hardcoded-text-only:
  source?: { file: string, fieldPath: string };  // present when content-index has a match
  requiresManualSelection?: boolean;             // true when signalCount < 2 OR (kind === 'hardcoded-text' AND no source)
}

<!-- Element selection ladder — apply in this order; first match claims the element so no element appears in two entries. -->

1. <img> elements with src starting /images/ → kind: 'image' or 'gallery-image' if inside data-gallery / data-room
2. Elements with data-i18n-html → kind: 'i18n-html'
3. Elements with data-i18n → kind: 'i18n-text'
4. Elements with data-i18n-placeholder → kind: 'i18n-text' (attr disambiguates downstream)
5. <h1>...<h6> elements that are NOT already claimed by an i18n attribute → kind: 'heading'
6. Text-bearing leaf elements (p, span, em, strong, li, a, button, blockquote) whose textContent.trim() is non-empty AND that are NOT inside an i18n-attributed ancestor → kind: 'hardcoded-text'
   - If contentIndex.get(normalizeText(text)) has an entry, set source to the first hit and requiresManualSelection: false
   - Otherwise set requiresManualSelection: true

<!-- ID computation — stable across builds. -->

id = sha1(kind + '|' + (i18nKey ?? '') + '|' + (imageRef ?? '') + '|' + (galleryAttrRaw ?? '') + '|' + (galleryIndex ?? '') + '|' + (domPath ?? '')).slice(0, 12)

Use node:crypto.createHash('sha1'). 12 hex chars is collision-safe at the per-route scale (<= 200 elements per CATALOG-07-equivalent cap).

<!-- signalCount rule — same as validate.ts:54 but adapted to the catalog field names. -->

A catalog entry has signalCount >= 2 when at least 2 of these are populated:
  - (i18nKey AND i18nAttr) counts as 1
  - imageRef counts as 1
  - (galleryAttrRaw AND Number.isInteger(galleryIndex) AND galleryIndex >= 0) counts as 1
  - (currentText length >= 25 AND nearestHeading) counts as 1
  - kind === 'hardcoded-text' AND source present counts as 1 (catalog-specific extension — the content-collection source acts as the 2nd anchor)

domPath alone NEVER counts. This is the rule v1.1 validate.ts pins.
