### Task 2 — Persist submission on success

Right before `showToast('ok', ...)` in the success branch, append the new submission to `localStorage`:

```js
function persistSubmission(data, payload) {
  try {
    var KEY = 'mar_feedback_recent_v1';
    var list = JSON.parse(localStorage.getItem(KEY) || '[]');
    if (!Array.isArray(list)) list = [];
    var routes = [];
    var editCount = 0;
    if (payload && Array.isArray(payload.edits)) {
      editCount = payload.edits.length;
      payload.edits.forEach(function (e) {
        var r = e && e.locator && e.locator.pageRoute;
        if (r && routes.indexOf(r) === -1) routes.push(r);
      });
    } else if (payload && payload.locator && payload.locator.pageRoute) {
      editCount = 1;
      routes.push(payload.locator.pageRoute);
    }
    list.unshift({
      issueNumber: data.issueNumber,
      issueUrl: data.issueUrl,
      summary: editCount + ' edit' + (editCount === 1 ? '' : 's') + (routes.length ? ' · ' + routes.join(', ') : ''),
      submittedAt: new Date().toISOString(),
      lastStage: 1,
      lastSub: null,
    });
    localStorage.setItem(KEY, JSON.stringify(list.slice(0, 20)));
  } catch (_) { /* localStorage unavailable: skip silently */ }
}
```

Invoke right after `if (res.ok && data.ok) {` block opens, before the postMessage. Pass `msg.payload` as the second arg.

### Task 4 — Rail render + poll loop

Inside the existing `<script is:inline>`, after `picker.addEventListener(...)` block:

```js
var KEY = 'mar_feedback_recent_v1';
var rail = document.getElementById('rail');
var railList = document.getElementById('railList');
var pollers = {};   // issueNumber → setTimeout id
var POLL_MS = 8000;

function loadRecent() {
  try { var v = JSON.parse(localStorage.getItem(KEY) || '[]'); return Array.isArray(v) ? v : []; }
  catch (_) { return []; }
}
function saveRecent(list) {
  try { localStorage.setItem(KEY, JSON.stringify(list.slice(0, 20))); } catch (_) {}
}
function relTime(iso) {
  var ms = Date.now() - new Date(iso).getTime();
  if (ms < 60000) return 'just now';
  if (ms < 3600000) return Math.floor(ms / 60000) + 'm ago';
  if (ms < 86400000) return Math.floor(ms / 3600000) + 'h ago';
  return Math.floor(ms / 86400000) + 'd ago';
}
function subLabel(sub, issueUrl) {
  if (sub === 'auto-merged' || sub === 'merged') return 'Merged';
  if (sub === 'needs-review') return "Needs Monty's review";
  if (sub === 'needs-client-reply') return 'Question for you → <a target="_blank" rel="noopener" href="' + issueUrl + '">open issue</a>';
  return '';
}
function stageLabel(stage) {
  return ['', 'Submitted', 'Reviewing', 'PR opened', 'Merged', 'Live'][stage] || '';
}

function renderRail() {
  var list = loadRecent();
  if (!list.length) { rail.hidden = true; return; }
  rail.hidden = false;
  railList.innerHTML = '';
  list.forEach(function (item) {
    var li = document.createElement('li');
    li.className = 'rail__row' + (item.lastStage === 5 ? ' is-live' : '');
    var bar = '';
    for (var i = 1; i <= 5; i++) {
      var cls = 'rail__bar__seg';
      if (i < item.lastStage) cls += ' is-done';
      else if (i === item.lastStage) cls += ' is-current';
      bar += '<div class="' + cls + '" title="' + stageLabel(i) + '"></div>';
    }
    var subHtml = '';
    if (item.lastStage === 5) {
      subHtml = '<div class="rail__row__sub">✓ Live · ' + relTime(item.submittedAt) + '</div>';
    } else if (item.lastSub) {
      subHtml = '<div class="rail__row__sub">' + subLabel(item.lastSub, item.issueUrl) + '</div>';
    } else {
      subHtml = '<div class="rail__row__sub">' + stageLabel(item.lastStage) + '…</div>';
    }
    li.innerHTML =
      '<div class="rail__row__head">' +
        '<a target="_blank" rel="noopener" href="' + item.issueUrl + '">#' + item.issueNumber + '</a>' +
        '<span class="when">' + relTime(item.submittedAt) + '</span>' +
      '</div>' +
      '<div class="rail__row__summary">' + item.summary + '</div>' +
      '<div class="rail__bar">' + bar + '</div>' +
      subHtml;
    railList.appendChild(li);
  });
}

function isTerminal(item) {
  if (item.lastStage === 5) return true;
  if (item.lastStage === 4 && (item.lastSub === 'needs-review' || item.lastSub === 'needs-client-reply')) return true;
  return false;
}

function pollOne(num) {
  if (pollers[num]) return;
  var loop = function () {
    fetch('/api/feedback/status/' + encodeURIComponent(num), { credentials: 'include' })
      .then(function (r) { return r.ok ? r.json() : null; })
      .then(function (data) {
        if (!data) return;
        var list = loadRecent();
        var idx = list.findIndex(function (x) { return String(x.issueNumber) === String(num); });
        if (idx === -1) return;
        list[idx].lastStage = data.stage;
        list[idx].lastSub = data.sub;
        saveRecent(list);
        renderRail();
        if (isTerminal(list[idx])) { clearInterval(pollers[num]); delete pollers[num]; }
      })
      .catch(function () { /* network blip: next tick will retry */ });
  };
  loop();
  pollers[num] = setInterval(loop, POLL_MS);
}

function startAllPollers() {
  loadRecent().forEach(function (item) { if (!isTerminal(item)) pollOne(item.issueNumber); });
}

renderRail();
startAllPollers();
```
