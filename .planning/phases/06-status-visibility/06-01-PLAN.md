# Plan 06-01 — Status endpoint + shared types

## Goal

Add the auth-gated `/api/feedback/status/[issueNumber]` endpoint that rolls GitHub issue + PR + Vercel deploy state into a 5-stage code. Mirror existing API-route conventions (`prerender = false`, `checkAuth()` from `src/lib/auth.ts`, `import type { APIRoute }`).

| File | Action |
|------|--------|
| `src/lib/feedback-status.ts` | NEW — shared types + pure stage-resolver function |
| `src/pages/api/feedback/status/[issueNumber].ts` | NEW — Astro API route, server-side memo cache |

## Tasks

### Task 1 — `src/lib/feedback-status.ts` (shared types + resolver)

Pure module (no top-level I/O). Exports:

```ts
export type FeedbackStage = 1 | 2 | 3 | 4 | 5;
export type FeedbackSub =
  | 'auto-merged' | 'needs-review' | 'needs-client-reply'
  | 'merged' | 'open' | null;

export interface FeedbackStatusInput {
  issueLabels: string[];
  issueHasPrLink: boolean;
  prFromComments: { url: string; number: number } | null;
  pr: { state: 'open' | 'closed'; merged: boolean; mergeSha: string | null; url: string; number: number } | null;
  vercelDeployState: 'READY' | 'BUILDING' | 'ERROR' | 'CANCELED' | null;
}

export interface FeedbackStatusOutput {
  stage: FeedbackStage;
  sub: FeedbackSub;
  prUrl: string | null;
  prNumber: number | null;
  mergeSha: string | null;
  deployUrl: string | null;
  updatedAt: string; // ISO 8601
}

export function resolveStage(input: FeedbackStatusInput): Pick<FeedbackStatusOutput,
  'stage' | 'sub' | 'prUrl' | 'prNumber' | 'mergeSha'>;
```

Resolver logic (deterministic, table-tested in Task 3):

1. If `pr === null && !issueHasPrLink && !prFromComments` → stage 2 if issue labels include `client-feedback`, else stage 1 (and `sub: null`).
2. If we have a PR but `pr.state === 'open' && !pr.merged` → stage 3 with `sub: 'open'`. If issue labels include `needs-review` → bump `sub` to `'needs-review'` (stage 4). If labels include `needs-client-reply` → `sub: 'needs-client-reply'` (stage 4).
3. If `pr.merged` → stage 4 with `sub: 'auto-merged'` if labels include `auto-merged`, else `'merged'`.
4. If stage 4 AND `pr.mergeSha` AND `vercelDeployState === 'READY'` → stage 5 with `sub: null`.
5. If stage 4 AND `vercelDeployState` is `null` (missing token) or `'BUILDING'` → stay at stage 4 with the sub-label preserved.

### Task 2 — `src/pages/api/feedback/status/[issueNumber].ts`

```ts
import type { APIRoute } from 'astro';
import { checkAuth } from '../../../../lib/auth';
import { resolveStage, type FeedbackStatusOutput } from '../../../../lib/feedback-status';

export const prerender = false;

const CACHE_TTL_MS = 5000;
type CacheEntry = { at: number; value: FeedbackStatusOutput };
const cache = new Map<string, CacheEntry>();

export const GET: APIRoute = async ({ params, request }) => {
  if (!(await checkAuth(request))) {
    return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401, headers: { 'Content-Type': 'application/json' } });
  }
  const num = String(params.issueNumber ?? '').trim();
  if (!/^\d+$/.test(num)) {
    return new Response(JSON.stringify({ error: 'Invalid issue number' }), { status: 400, headers: { 'Content-Type': 'application/json' } });
  }
  const cached = cache.get(num);
  if (cached && Date.now() - cached.at < CACHE_TTL_MS) {
    return new Response(JSON.stringify(cached.value), { status: 200, headers: { 'Content-Type': 'application/json' } });
  }

  // Source: issue (labels, pull_request link)
  const repo = (import.meta.env.GITHUB_REPO || '').trim();
  const tok = (import.meta.env.GITHUB_TOKEN || '').trim();
  if (!repo || !tok) {
    return new Response(JSON.stringify({ error: 'Server misconfigured' }), { status: 500, headers: { 'Content-Type': 'application/json' } });
  }
  const ghHeaders = { Authorization: `Bearer ${tok}`, Accept: 'application/vnd.github+json', 'X-GitHub-Api-Version': '2022-11-28' };

  const issueRes = await fetch(`https://api.github.com/repos/${repo}/issues/${num}`, { headers: ghHeaders });
  if (!issueRes.ok) {
    return new Response(JSON.stringify({ error: `Issue lookup failed (${issueRes.status})` }), { status: issueRes.status === 404 ? 404 : 502, headers: { 'Content-Type': 'application/json' } });
  }
  const issue = await issueRes.json();
  const issueLabels: string[] = Array.isArray(issue.labels) ? issue.labels.map((l: any) => (typeof l === 'string' ? l : l?.name)).filter(Boolean) : [];
  const issueHasPrLink = !!(issue.pull_request && issue.pull_request.url);

  // Fallback: scan comments for a PR URL the Action posted
  let prFromComments: { url: string; number: number } | null = null;
  if (!issueHasPrLink) {
    const cRes = await fetch(`https://api.github.com/repos/${repo}/issues/${num}/comments?per_page=100`, { headers: ghHeaders });
    if (cRes.ok) {
      const comments = await cRes.json();
      for (const c of comments) {
        const m = String(c.body || '').match(new RegExp(`https://github.com/${repo}/pull/(\\d+)`));
        if (m) { prFromComments = { url: m[0], number: parseInt(m[1], 10) }; break; }
      }
    }
  }

  // Source: PR (state, merged, merge_commit_sha)
  let pr: NonNullable<Parameters<typeof resolveStage>[0]['pr']> | null = null;
  const prNumber = issueHasPrLink
    ? parseInt(String(issue.pull_request.url).split('/').pop() || '', 10)
    : (prFromComments ? prFromComments.number : NaN);
  if (Number.isFinite(prNumber)) {
    const prRes = await fetch(`https://api.github.com/repos/${repo}/pulls/${prNumber}`, { headers: ghHeaders });
    if (prRes.ok) {
      const p = await prRes.json();
      pr = {
        state: p.state,
        merged: !!p.merged,
        mergeSha: p.merge_commit_sha || null,
        url: p.html_url,
        number: p.number,
      };
    }
  }

  // Source: Vercel deployment for merge SHA (best-effort)
  let vercelDeployState: 'READY' | 'BUILDING' | 'ERROR' | 'CANCELED' | null = null;
  let deployUrl: string | null = null;
  const vercelTok = (import.meta.env.VERCEL_TOKEN || '').trim();
  const vercelProjectId = (import.meta.env.VERCEL_PROJECT_ID || 'prj_RIo7DcBIYysCuz9zhnI71u1Ifrb4').trim();
  if (vercelTok && pr?.merged && pr.mergeSha) {
    const vurl = `https://api.vercel.com/v6/deployments?projectId=${vercelProjectId}&limit=20&target=production`;
    const vRes = await fetch(vurl, { headers: { Authorization: `Bearer ${vercelTok}` } });
    if (vRes.ok) {
      const v = await vRes.json();
      const match = (v.deployments || []).find((d: any) => d.meta?.githubCommitSha === pr!.mergeSha);
      if (match) {
        vercelDeployState = match.readyState || match.state || null;
        deployUrl = match.url ? `https://${match.url}` : null;
      }
    }
  }

  const resolved = resolveStage({ issueLabels, issueHasPrLink, prFromComments, pr, vercelDeployState });
  const out: FeedbackStatusOutput = {
    ...resolved,
    deployUrl,
    updatedAt: new Date().toISOString(),
  };
  cache.set(num, { at: Date.now(), value: out });
  return new Response(JSON.stringify(out), { status: 200, headers: { 'Content-Type': 'application/json' } });
};
```
