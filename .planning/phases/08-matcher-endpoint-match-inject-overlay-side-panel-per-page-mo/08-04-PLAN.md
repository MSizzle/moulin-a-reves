    function activateMode(mode) {
      var isMatch = mode === 'per-page';
      modeTabClick.classList.toggle('is-active', !isMatch);
      modeTabMatch.classList.toggle('is-active', isMatch);
      modeTabClick.setAttribute('aria-selected', !isMatch);
      modeTabMatch.setAttribute('aria-selected', isMatch);
      modeTabClick.setAttribute('tabindex', isMatch ? '-1' : '0');
      modeTabMatch.setAttribute('tabindex', isMatch ? '0' : '-1');
      modePaneClick.hidden = isMatch;
      modePaneMatch.hidden = !isMatch;
      // Update URL hash without scrolling (D-08): replace, not push
      var newHash = '#mode=' + (isMatch ? 'per-page' : 'per-element');
      if (location.hash !== newHash) history.replaceState(null, '', location.pathname + location.search + newHash);
    }
    function readModeFromHash() {
      var m = (location.hash || '').match(/mode=(per-element|per-page)/);
      return m ? m[1] : 'per-element'; // default per D-07
    }
    modeTabClick.addEventListener('click', function () { activateMode('per-element'); });
    modeTabMatch.addEventListener('click', function () { activateMode('per-page'); });
    // Arrow-key tab navigation (ARIA tabs pattern, UI-SPEC.md §1)
    [modeTabClick, modeTabMatch].forEach(function (tab) {
      tab.addEventListener('keydown', function (ev) {
        if (ev.key === 'ArrowLeft' || ev.key === 'ArrowRight') {
          ev.preventDefault();
          activateMode(tab === modeTabClick ? 'per-page' : 'per-element');
          (tab === modeTabClick ? modeTabMatch : modeTabClick).focus();
        }
      });
    });
    // Initial activation from hash
    activateMode(readModeFromHash());
